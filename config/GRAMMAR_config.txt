'WORD'          :

                |       server_BLOCK

 global_BLOCK   :       'global' '{' g_SETTING '}'
 g_SETTING      :       g_SETTING g_OPTION ';'
                |       g_OPTION ';'
 g_OPTION       :       g_Key g_Value
 g_Key          :       WORD
 g_Value        :       g_Value path
                |       g_Value WORD
                |       path
                |       WORD


 server_BLOCK   :       'server' {' s_SETTING location_BLOCK '}'
 s_SETTING      :       s_SETTING s_OPTION ';'
                |       s_OPTION ';'
 s_OPTION       :       s_Key s_Value
 s_Key          :       WORD
 s_Value        :       s_Value path
                |       s_Value WORD
                |       path
                |       WORD


 location_BLOCK :       'location' path '{' l_SETTING '}'
 l_SETTING      :       l_SETTING l_OPTION ';'
                |       l_OPTION ';'
 l_OPTION       :       l_Key l_Value
 l_Key          :       WORD
 l_Value        :       l_Value path
                |       l_Value WORD
                |       path
                |       WORD
  
 path           :       WORD
